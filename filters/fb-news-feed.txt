! news feed: sponsored
!- need both rules (fb sometimes insert an extra span ...)
!facebook.com###facebook div[role="feed"] div > span > span > a > span[aria-label="Sponsored"]:upward(div[role="feed"] > div)
! :: revised rule for hiding sponsored posts; based on html structure;
! news feed: sponsored (news feed column only - does not apply to group feeds)
! -- need the data-pagelet to work properly.
facebook.com##div[role="feed"] > div[data-pagelet^="FeedUnit"] div > span > span > a > span:upward(div[data-pagelet^="FeedUnit"])
! -- another "sponsored", but with slightly different structure - have to find the works this time round...
facebook.com##div[role="feed"] > div[data-pagelet^="FeedUnit"] div > span > span > div:has-text(/Sponsored · Paid for by/i):upward(div[data-pagelet^="FeedUnit"])
!
! news feed: people you may know
facebook.com###facebook div[role="feed"] div > span:has-text(/People you may know/i):upward(div[role="feed"] > div)
!
! news feed: videos just for you 
facebook.com###facebook div[role="feed"] div > span:has-text(/Videos just for you/i):upward(div[role="feed"] > div)
!
! news feed: suggested for you  
facebook.com###facebook div[role="feed"] div > span:has-text(/suggested for you/i):upward(div[role="feed"] > div)
!
! right-hand column: sponsored
facebook.com###facebook div[data-pagelet="RightRail"] > div > div > div h3 span:has-text(/Sponsored/i):upward(div[data-pagelet="RightRail"] > div)
facebook.com###facebook div[data-pagelet="RightRail"] > div > div > div h3 span:has-text(/Sponsorisé/i):upward(div[data-pagelet="RightRail"] > div)
